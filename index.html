                    void mergesort(vector >& a, int left, int right) {
                        if (left < right) {
                            int medium = (left + right) / 2;
                            mergesort(a, left, medium);     //сортируем левую часть
                            mergesort(a, medium + 1, right); //сортируем правую часть
                            merge(a, left, medium, right);      //сливаем
                        }
                    }
                    
                    void merge(std::vector >& a, int left, int medium, int right) {
                        int sizeL = medium - left + 1, sizeR = right - medium;
                        vector > L(sizeL);
                        vector > R(sizeR);
                        for (int i = 0; i < sizeL; ++i) {
                            L[i].resize(2);
                            for (int j = 0; j < 2; ++j)
                                L[i][j] = a[i + left][j];
                        }
                    
                        for (int i = 0; i < sizeR; ++i) {
                            R[i].resize(2);
                            for (int j = 0; j < 2; ++j)
                                R[i][j] = a[medium + 1 + i][j];
                        }
                    
                        int i = 0, j = 0;
                        for (int currentIndex = left; currentIndex <= right; ++currentIndex)       //пока есть эллементы хотя бы в 1 массиве сливаем
                            if (i < sizeL && j < sizeR)      //если остались эллементы в обоих массивах сравниваем их
                                if (L[i][0] <= R[j][0]) {
                                    a[currentIndex][0] = L[i][0];
                                    a[currentIndex][1] = L[i][1];
                                    ++i;
                                }
                                else {
                                    a[currentIndex][0] = R[j][0];
                                    a[currentIndex][1] = R[j][1];
                                    ++j;
                                }
                            else {                     //если в 1 из массивов закончились эллемены дописываем эллементы другого массива
                                if (i < sizeL) {
                                    a[currentIndex][0] = L[i][0];
                                    a[currentIndex][1] = L[i][1];
                                    ++i;
                                }
                                else
                                    if (j < sizeR) {
                                        a[currentIndex][0] = R[j][0];
                                        a[currentIndex][1] = R[j][1];
                                        ++j;
                                    }
                            }
                    
                    }
